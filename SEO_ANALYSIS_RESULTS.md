### 1. Created Reusable SEO Composable
**File:** `composables/useSeoCanonical.ts`

```typescript
export const useSeoCanonical = (path: string) => {
  const { locale } = useI18n();
  const baseUrl = 'https://diktat.ai';

  const canonicalPath = computed(() => {
    // Default locale (de) â†’ no prefix
    if (locale.value === 'de') {
      return path.startsWith('/') ? path : `/${path}`;
    }
    // Other locales (en) â†’ add locale prefix
    return `/en${path.startsWith('/') ? path : `/${path}`}`;
  });

  const canonicalUrl = computed(() => `${baseUrl}${canonicalPath.value}`);

  return { canonicalUrl, canonicalPath, baseUrl };
};
```

**Benefits:**
- Centralized canonical URL logic
- Automatic locale handling
- Consistent implementation across all pages
- Always uses `diktat.ai` domain
